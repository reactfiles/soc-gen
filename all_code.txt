    <script>
        var count = 1;
        function addClickHandler(){
           // alert("lets create a box");
           var elm = document.createElement("div");
           elm.textContent = count;
           document.getElementById("container").appendChild(elm);
           count++;
        }
        function init(){
            document.getElementsByTagName("button")[0].addEventListener("click", addClickHandler)
        };
        document.addEventListener("DOMContentLoaded", init);
